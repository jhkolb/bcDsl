data {
  Seller: Identity
  HighestBidder: Identity
  HighestBid: Int
  Duration: Timespan
}

->(duration: Timespan) open {
    Seller = sender
    HighestBid = 0
    Duration = duration
}

open ->(bid: Int) open
requires [ bid > HighestBid ] {
    HighestBidder = sender
    HighestBid = bid
}

open -> closed
onlyAfter [ Duration ]
